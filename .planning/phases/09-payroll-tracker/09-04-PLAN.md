  <action>
    Create app/(portal)/payroll/PayrollClient.tsx as a 'use client' component.

    Props interface:
      PayrollClientProps {
        initialStaff: StaffPayrollRow[];
        initialTotal: number;
        initialMonth: string;  // ISO date "YYYY-MM-01"
      }

    State:
      - selectedMonth: string (useState, default = props.initialMonth)
      - staff: StaffPayrollRow[] (useState, default = props.initialStaff)
      - total: number (useState, default = props.initialTotal)
      - isNavigating: boolean (useState false) — controls Skeleton loading state during month change
      - pendingIds: Set<string> (useState new Set()) — tracks which row's "Mark as Paid" button is in pending state

    Month navigation helpers (inside component or as module-level utils):
      - toFirstOfMonth(date: Date): string — same helper used in page.tsx
      - navigateMonth(dir: 'prev' | 'next'): void
          const [y, m] = selectedMonth.split('-').map(Number)
          const d = new Date(y, m - 1 + (dir === 'next' ? 1 : -1), 1)
          setSelectedMonth(toFirstOfMonth(d))
          then call loadMonthData(toFirstOfMonth(d))
      - currentMonth: string = toFirstOfMonth(new Date())  (computed once per render or useMemo)
      - isAtCurrentMonth: boolean = selectedMonth >= currentMonth
      - formatMonthDisplay(iso: string): string
          const [y, m] = iso.split('-').map(Number)
          return new Date(y, m - 1, 1).toLocaleDateString('en-IN', { month: 'long', year: 'numeric' })
          // e.g. "June 2026"

    loadMonthData(month: string): async function
      - setIsNavigating(true)
      - const result = await getPayrollDataAction(month)
      - if result.error: toast.error(result.error)
      - else: setStaff(result.staff); setTotal(result.total)
      - setIsNavigating(false)

    handleMarkAsPaid(row: StaffPayrollRow): async function
      - setPendingIds(prev => new Set(prev).add(row.id))
      - const amountPaid = row.salary ?? 0
      - const result = await markAsPaidAction({ profileId: row.id, paymentMonth: selectedMonth, amountPaid })
      - setPendingIds(prev => { const s = new Set(prev); s.delete(row.id); return s })
      - if result.error: toast.error(result.error); return
      - if amountPaid === 0 and row.salary === null: toast.success('Marked as ₹0. Update salary on the Staff page first.')
      - else: toast.success(`Marked as paid — ${formatINR(amountPaid)}`)
      - reload current month data: loadMonthData(selectedMonth)

    formatINR(amount: number): string
      return new Intl.NumberFormat('en-IN', { style: 'currency', currency: 'INR', maximumFractionDigits: 0 }).format(amount)

    Layout (per UI-SPEC §Layout Contract):
      1. Page Header: flex row space-between items-center
         - Title "Payroll" (text-xl font-semibold)
         - Month Navigator: flex row gap-2 items-center
           - Button ghost ChevronLeft (aria-label="Previous month", min-h-[48px]) → navigateMonth('prev')
           - Month label span (text-xl font-semibold, min-w-[140px] text-center) → formatMonthDisplay(selectedMonth)
           - Button ghost ChevronRight (aria-label="Next month", min-h-[48px], disabled={isAtCurrentMonth}) → navigateMonth('next')

      2. Summary Card (Card component, muted bg, full width):
         - Label: "Total Paid This Month" (text-xs text-muted-foreground)
         - Value: formatINR(total) (text-3xl font-semibold)

      3. Staff Payment Table (Table component):
         TableHeader: Name | Role | Monthly Salary | Status | (empty action header)
         TableBody:
           - if isNavigating: show 5 Skeleton rows (each cell has a <Skeleton className="h-4 w-full" />)
           - if !isNavigating and staff.length === 0: single row with colspan=5, "No active staff found" + "Add staff on the Staff page before running payroll."
           - else: staff.map(row =>
               TableRow key={row.id}:
                 - Name cell: row.full_name ?? '—'  (text-sm)
                 - Role cell: row.role  (text-sm text-muted-foreground)
                 - Salary cell: row.salary !== null ? formatINR(row.salary) : '—'  (text-sm)
                 - Status cell: Badge — if row.isPaid: variant="default" "Paid" else: variant="secondary" "Unpaid"
                 - Action cell:
                   - if row.isPaid: span text-xs text-muted-foreground "Paid on [Day Mon]"
                     format paidAt: new Date(row.paidAt!).toLocaleDateString('en-IN', { day: 'numeric', month: 'short' })
                     e.g. "Paid on 13 Jun"
                   - if !row.isPaid: Button default variant size sm
                     disabled={pendingIds.has(row.id)}
                     onClick={() => handleMarkAsPaid(row)}
                     content: if pendingIds.has(row.id): <Loader2 className="h-3 w-3 animate-spin" /> "Marking..."
                     else: "Mark as Paid"
             )

    Spacing (per UI-SPEC §Spacing Scale):
      - Page header to content: mt-8 (xl = 32px)
      - Summary card to table: mt-6 (lg = 24px)
      - Card padding: p-4 (md = 16px)
